\begin{frame}
  \frametitle{矩方程}

  \begin{itemize}
    \item Hamiltonian
    \begin{align}
      H_{\T} = h_{\tL} + h_{\tR} + \hat F_{\tL}\hat F_{\tR}
    \end{align}
    \item 耗散子分解
    \begin{align}
      \hat F_{\alpha = \tL,\tR} = \sum_k \hat f_{\alpha k}
    \end{align}
    \item 定义
    \begin{align}
      f_{\alpha k}(t) \equiv {\rm Tr}[\hat f_{\alpha k}\rho_{\T}(t)], \quad \sigma_{\alpha k,\alpha'k'}(t) \equiv {\rm Tr}[(\hat f_{\alpha k}\hat f_{\alpha' k'})^\circ\rho_{\T}(t)] = \sigma_{\alpha' k',\alpha k}(t)
    \end{align}
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{矩方程}

  \begin{itemize}
    \item 均值的方程
    \begin{align}
      \dot f_{\tL k}(t) &= {\rm Tr}[\hat f_{\tL k}\dot\rho_{\T}(t)] = -\gamma_{\tL k}f_{\tL k}(t) -i (\eta_{\tL k} - \eta_{\tL \bar k}^*)\sum_{k'}f_{\tR k'}(t)\\
      \dot f_{\tR k}(t) &= {\rm Tr}[\hat f_{\tR k}\dot\rho_{\T}(t)] = -\gamma_{\tR k}f_{\tR k}(t) -i (\eta_{\tR k} - \eta_{\tR \bar k}^*)\sum_{k'}f_{\tL k'}(t)
    \end{align}
    由初值$f_{\alpha k}(0) = 0$可知
    \begin{align}
      f_{\alpha k}(t) = 0
    \end{align}
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{矩方程}

  \begin{itemize}
    \item 方差的方程
    \begin{align}
      \dot \sigma_{\tL k,\tL k'} &= -(\gamma_{\tL k} + \gamma_{\tL k'})\sigma_{\tL k,\tL k'} -i (\eta_{\tL k} - \eta_{\tL \bar k}^*)\sum_{j}\sigma_{\tL k',\tR j}\nl
      &\quad \,-i (\eta_{\tL k'} - \eta_{\tL \bar k'}^*)\sum_{j}\sigma_{\tL k,\tR j} \\ 
      \dot \sigma_{\tR k,\tR k'} &= -(\gamma_{\tR k} + \gamma_{\tR k'})\sigma_{\tR k,\tR k'} -i (\eta_{\tR k} - \eta_{\tR \bar k}^*)\sum_{j}\sigma_{\tR k',\tL j}\nl
      &\quad \,-i (\eta_{\tR k'} - \eta_{\tR \bar k'}^*)\sum_{j}\sigma_{\tR k,\tL j} \\ 
      \dot \sigma_{\tL k,\tR k'} &= -(\gamma_{\tL k} + \gamma_{\tR k'})\sigma_{\tL k,\tR k'} -i (\eta_{\tL k}\eta_{\tR k'} - \eta_{\tL \bar k}^*\eta_{\tR \bar k'}^*) \nl
      &\quad \,-i (\eta_{\tL k} - \eta_{\tL \bar k}^*)\sum_{j}\sigma_{\tR k',\tR j} -i (\eta_{\tR k'} - \eta_{\tR \bar k'}^*)\sum_{j}\sigma_{\tL k,\tL j}
    \end{align}
  \end{itemize}

\end{frame}

  \frametitle{运动方程}

  \begin{align}
    \dot\rho_{\bf n}^{(n)}(t) &= -\bigg( H_{\tS}^\times + \sum_k n_k\gamma_k \bigg)\rho_{\bf n}^{(n)}(t) -i \hat Q_1\rho_{\bf n}^{(n)}(t; \hat F_1^{r_1}) -i \hat Q_2\rho_{\bf n}^{(n)}(t; \hat F_2^{r_2}) \cdots  \nl
    &\quad \,-i \hat Q_{12}\rho_{\bf n}^{(n)}(t; \hat F_1^{s_1}\hat F_2^{s_2}) + \cdots
  \end{align}

  \begin{align}
    \hat F_a = \sum_k p_{ak}\hat f_k
  \end{align}

  \begin{align}
    \rho_{\bf n}^{(n)}(t; \hat F_a) = \sum_k p_{ak} \bigg[ \sqrt{n_{k} + 1}\sqrt{||\eta_k||} \rho_{{\bf n}_{k}^+}^{(n+1)}(t) + \frac{\sqrt{n_k} \eta_k}{\sqrt{||\eta_k||}} \rho_{{\bf n}_{k}^-}^{(n-1)}(t)\bigg]
  \end{align}

  \begin{align}
    \rho_{\bf n}^{(n)}(t; \hat \Phi_a) = \sum_k -\gamma_{k}p_{ak} \bigg[ \sqrt{n_{k} + 1}\sqrt{||\eta_k||} \rho_{{\bf n}_{k}^+}^{(n+1)}(t) - \frac{\sqrt{n_k} \eta_k}{\sqrt{||\eta_k||}} \rho_{{\bf n}_{k}^-}^{(n-1)}(t)\bigg]
  \end{align}

  \begin{align}
    \rho_{\bf n}^{(n)}(t; \hat F_a^2) = \sum_k p_{ak} \bigg[ \sqrt{n_{k} + 1}\sqrt{||\eta_k||} \rho_{{\bf n}_{k}^+}^{(n+1)}(t; \hat F_a) + \frac{\sqrt{n_k} \eta_k}{\sqrt{||\eta_k||}} \rho_{{\bf n}_{k}^-}^{(n-1)}(t; \hat F_a)\bigg]
  \end{align}
